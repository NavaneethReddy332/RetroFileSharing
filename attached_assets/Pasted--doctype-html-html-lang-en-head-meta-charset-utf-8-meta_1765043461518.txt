  <script>
    // Tabs highlight
    const tabs = document.querySelectorAll('.tab');
    const hl = document.getElementById('highlight');
    function placeHL(el){
      const r = el.getBoundingClientRect();
      const parent = el.parentElement.getBoundingClientRect();
      hl.style.left = (r.left - parent.left) + 'px';
      hl.style.width = r.width + 'px';
    }
    // init on first tab
    placeHL(tabs[0]);
    tabs.forEach(t=>t.addEventListener('click', e=>{
      tabs.forEach(x=>x.classList.remove('active'));
      e.currentTarget.classList.add('active');
      placeHL(e.currentTarget);
    }));

    // Drag & drop
    const drop = document.getElementById('drop');
    const fileInput = document.getElementById('file');
    const filesEl = document.getElementById('files');
    const clearBtn = document.getElementById('clear-list');

    function addFilePreview(file){
      const row = document.createElement('div');
      row.className='file-row';
      const left = document.createElement('div'); left.style.flex='1';
      const name = document.createElement('div'); name.className='file-name'; name.textContent = file.name;
      const meta = document.createElement('div'); meta.className='file-meta'; meta.textContent = (file.size/1024/1024>1? (file.size/1024/1024).toFixed(2) + ' MB' : (file.size/1024).toFixed(0) + ' KB') + ' ‚Äî ready';
      left.appendChild(name); left.appendChild(meta);
      const btns = document.createElement('div'); btns.style.width='90px'; btns.style.textAlign='right';
      const rem = document.createElement('button'); rem.className='btn'; rem.textContent='Remove'; rem.style.padding='6px 8px';
      rem.onclick = ()=>row.remove();
      btns.appendChild(rem);
      row.appendChild(left); row.appendChild(btns);
      filesEl.appendChild(row);
    }

    fileInput.addEventListener('change', e=>{
      Array.from(e.target.files).forEach(f=>addFilePreview(f));
    });

    drop.addEventListener('click', ()=>fileInput.click());
    drop.addEventListener('dragover', e=>{e.preventDefault();drop.classList.add('dragover')});
    drop.addEventListener('dragleave', e=>{drop.classList.remove('dragover')});
    drop.addEventListener('drop', e=>{
      e.preventDefault();drop.classList.remove('dragover');
      const dt = e.dataTransfer; if(!dt) return; Array.from(dt.files).forEach(f=>addFilePreview(f));
    });

    clearBtn.addEventListener('click', ()=>{filesEl.innerHTML=''});

    // Refresh mock
    document.getElementById('refresh').addEventListener('click', ()=>{
      const el = document.querySelector('.panel .footer-label');
      el.textContent = 'Refreshed ' + new Date().toLocaleTimeString();
      setTimeout(()=>{el.textContent='Tip: drag files into the left panel to start'},2200);
    });

    // small UX: animate highlight initially when page loads
    window.addEventListener('resize', ()=>{
      const active = document.querySelector('.tab.active'); if(active) placeHL(active);
    });
  </script>
